LIBRARY IEEE;
USE IEEE.STD_LOGIC_1164.ALL;
ENTITY D_FF IS
	PORT (a,b,c,d,CLK,CE: IN STD_LOGIC; co : OUT STD_LOGIC);
END ENTITY D_FF

ARCHITECTURE dff1 OF D_FF IS
BEGIN
 CLK<=c and d;
 PROCESS(CLK)
  BEGIN
   IF(CLK'event and CLK='1') THEN
    IF CE='1' THEN
     co<=a and b;	
    END IF
   END IF
 END PROCESS	
END ARCHITECTURE fh1;
